from reqable import *

# RevenueCat 相关
Modify_Token = False    # 默认不修改fetch_token
RC_HOST = "api.revenuecat.com"
RC_RECEIPTS_PATH = "/v1/receipts"

def onRequest(context, request):
    # 1. 处理 OpenAI 埋点请求
    if OAI_HOST in context.host and OAI_TRACKING_PATH in request.path:
        try:
            request.body.jsonify()
            payload = request.body.payload
            if "batch" in payload:
                for item in payload["batch"]:
                    if "properties" in item and "offer_ids" in item["properties"]:
                        item["properties"]["offer_ids"] = "oai-chatgpt-plus-1m-1999:plus-1-month-free-trial"
                context.highlight = Highlight.yellow
                context.comment = "OAI: offer_ids modified"
        except:
            pass
            
    return request

def onResponse(context, response):
    # 获取请求路径用于匹配
    req_path = response.request.path

    # 1. 处理 OpenAI 账号/支付校验
    if OAI_HOST in context.host:
        if any(path in req_path for path in OAI_RESPONSE_PATHS):
            try:
                response.body.jsonify()
                data = response.body.payload
                # 遍历修改所有账号信息
                accounts = data.get("accounts", {})
                for acc_id in accounts:
                    modify_account_info(accounts[acc_id])
                if "default" in data:
                    modify_account_info(data["default"])
                
                context.highlight = Highlight.green
                context.comment = "OAI: Plus Patched"
            except:
                pass

    # 2. 处理 RevenueCat 收据校验
    elif Modify_Token and RC_HOST in context.host and RC_RECEIPTS_PATH in req_path:
        try:
            response.body.jsonify()
            data = response.body.payload
            # 替换 fetch_token (通常在根目录或 subscriber 字段中)
            if "fetch_token" in data:
                data["fetch_token"] = NEW_FETCH_TOKEN
            
            # 如果 RevenueCat 返回的数据里包含 subscriber，顺便确保权益激活
            if "subscriber" in data:
                sub = data["subscriber"]
                if "fetch_token" in sub:
                    sub["fetch_token"] = NEW_FETCH_TOKEN
            
            context.highlight = Highlight.blue
            context.comment = "RC: fetch_token replaced"
        except:
            pass

    return response

def modify_account_info(acc_obj):
    """修改 OpenAI 账号对象"""
    acc_obj["eligible_promo_campaigns"] = PROMO_CAMPAIGN
